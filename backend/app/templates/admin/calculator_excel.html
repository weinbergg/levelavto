Расширенный сценарий — выборочный пересчёт через консоль
Если конфиг затрагивает только определённый возраст / регион — можно ограничить пересчёт. Сначала --dry-run покажет количество и время без запуска. 
docker compose exec -T web python -m backend.app.scripts.recalc_calc_cache \
  --region EU --age-min 3 --age-max 5 --dry-run     # оценка без запуска

docker compose exec -T web python -m backend.app.scripts.recalc_calc_cache \
  --region EU --age-min 3 --age-max 5               # реальный пересчёт
